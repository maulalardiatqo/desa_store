#include <WiFi.h>
#include <HTTPClient.h>
#include <SPI.h>
#include <MFRC522.h>

#define SS_PIN 21
#define RST_PIN 22
MFRC522 mfrc522(SS_PIN, RST_PIN);

// Ganti dengan WiFi kamu
const char* ssid = "Dheardi.net";
const char* password = "dheardi3323";

// Ganti IP sesuai server CI kamu
const char* host = "http://192.168.0.105/absensi";

void setup() {
  Serial.begin(9600);
  SPI.begin();
  mfrc522.PCD_Init();

  Serial.print("Menghubungkan ke WiFi");
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nTerhubung ke WiFi!");
}

void loop() {
  // Cek apakah ada kartu yang baru ditap
  if (!mfrc522.PICC_IsNewCardPresent() || !mfrc522.PICC_ReadCardSerial()) return;

  String uid = "";
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    if (mfrc522.uid.uidByte[i] < 0x10) uid += "0";
    uid += String(mfrc522.uid.uidByte[i], HEX);
  }
  uid.toUpperCase();

  Serial.println("UID Terdeteksi: " + uid);

  kirimUID(uid);

  delay(3000); // Hindari spam kirim UID
}

void kirimUID(String uid) {
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    String url = String(host) + "/admin/set_latest_uid";
    String data = "uid=" + uid;

    http.begin(url);
    http.addHeader("Content-Type", "application/x-www-form-urlencoded");

    int httpCode = http.POST(data);
    Serial.print("HTTP Code: ");
    Serial.println(httpCode);

    String response = http.getString();
    Serial.print("Response Server: ");
    Serial.println(response);

    if (httpCode > 0 && httpCode == HTTP_CODE_OK) {
      Serial.println("UID berhasil dikirim ke server.");
    } else {
      Serial.println("Gagal kirim UID ke server.");
    }

    http.end();
  } else {
    Serial.println("WiFi tidak terhubung.");
  }
}
